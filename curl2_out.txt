            (function() {
                // 1. Clean up previous listener if it exists (prevents stacking on Turbo reload)
                if (window.NozConfirmHandler) {
                    document.removeEventListener('click', window.NozConfirmHandler, true);
                }

                // 2. Define new handler
                window.NozConfirmHandler = function(e) {
                    const trigger = e.target.closest('[data-confirm]');
                    if (trigger) {
                        e.preventDefault();
                        e.stopPropagation();
                        e.stopImmediatePropagation();

                        // Always look up elements dynamically to get the current ones in the DOM
                        const modal = document.getElementById('confirmation-modal');
                        const messageEl = document.getElementById('modal-message');
                        const confirmBtn = document.getElementById('modal-confirm-btn');
                        const cancelBtn = document.getElementById('modal-cancel-btn');

                        if (!modal) return;

                        // Setup Modal Content
                        messageEl.textContent = trigger.dataset.confirm;
                        modal.classList.remove('hidden');

                        // Setup Actions
                        confirmBtn.onclick = function() {
                            if (trigger.tagName === 'A') {
                                if (typeof Turbo !== 'undefined') {
                                    Turbo.visit(trigger.href);
                                } else {
                                    window.location.href = trigger.href;
                                }
                            } else if (trigger.tagName === 'BUTTON' && trigger.type === 'submit') {
                                const form = trigger.closest('form');
                                if (form) form.submit();
                            }
                            modal.classList.add('hidden');
                        };

                        cancelBtn.onclick = function() {
                            modal.classList.add('hidden');
                        };

                        modal.onclick = function(e) {
                            if (e.target === modal.querySelector('.fixed.inset-0')) {
                                modal.classList.add('hidden');
                            }
                        };
                    }
                };

                // 3. Attach new listener (Capture Phase)
                document.addEventListener('click', window.NozConfirmHandler, true);
            })();
